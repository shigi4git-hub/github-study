＝＝＝ここまでの主なコマンド＝＝＝
mkdir github-workspace　※作業用フォルダを作成
cd github-workspace　※フォルダに移動
git clone https://github.com/shigi4git-hub/github-study.git　※リポジトリをクローン（ローカルフォルダにファイル取得）
echo "これはテストファイルです" > test.txt　※テキストファイルの生成
git add test.txt　※作成したテキストファイルをローカルに追加(add)。
git commit -m "テストファイルを追加"　※テキスト追加をMDファイルに追記(commit)。
git push origin main　※変更をGitHubにアップロード（push）
（Git-Hubで、テキスト編集）

＝＝この後のコマンド＝＝
git pull origin main（ローカルに最新情報を保存）

＝＝作業ログ用・テキスト更新した際のコマンド＝＝
git branch　※現在地を確認。mainでない場合、mainに戻る。
git checkout main　※（作業ブランチから）mainブランチに戻る
git pull origin main　※GitHubの最新状態をローカルに反映
git checkout -b add-commands-list　※コマンド一覧を追記するための新しいブランチを作成（直接作業による環境破壊防止のため、新規ブランチで実施がお作法とのこと）
echo "　※テキストファイルを更新（他コマンド等あり）

＝＝＝Issues・Pull Requestで使用したコマンド加筆＝＝＝

git checkout -b update-readme

# 3. 確認
cat README.md

git add README.md
# 5. 変更を記録
git commit -m "README.mdに学習内容を追記 #1"

# ※Web上でMergeボタンをクリック

# 不要ブランチ削除の手順
git branch                      # 現在のブランチ一覧と現在地を表示
git checkout main               # mainブランチに移動
git pull origin main            # GitHubの最新状態を取得・反映
git branch -d ブランチ名        # 不要なブランチを削除
git branch                      # 削除されたか確認
□Git Flow
# ===== 1. 現在地確認 =====
git branch
# 結果：* main
# ===== 2. developブランチ作成・移動 =====
git checkout -b develop
# 結果：Switched to a new branch 'develop'
# プロンプト：(main) → (develop)
# ===== 3. developをGitHubにpush =====
git push origin develop
# 結果：GitHub上にdevelopブランチ作成
# ===== 4. 現在地確認（念のため） =====
git branch
# 結果：
#   * develop
#     main
# ===== 5. featureブランチ作成・移動 =====
git checkout -b feature/add-gitflow-notes
# 結果：Switched to a new branch 'feature/add-gitflow-notes'
# プロンプト：(develop) → (feature/add-gitflow-notes)
# ===== 6. test.txtに追記（これから実行） =====
＝＝＝Git Flowのブランチ構造＝＝＝
main（本丸・本番環境）
  └─ develop（開発拠点・開発環境）
       └─ feature/機能名（作業場・新機能開発）
【ブランチの役割】
main：本番公開中のバージョン（絶対に守る）
develop：開発中の最新版（次のリリース準備）
feature：個別の機能開発（作業用）
【作業の流れ】
1. feature で作業
2. develop にマージ
3. develop から main にマージ（本番公開）
# 7. 変更を確認
cat test.txt
# 8. 変更をステージング
git add test.txt
# 9. コミット
git commit -m "Git Flowのメモを追加"
# 10. GitHubにpush
git push origin feature/add-gitflow-notes
# 11. GitHubでPull Request作成（Web操作）
#     feature/add-gitflow-notes → develop
# 12. マージ後、developに戻る
git checkout develop
# 13. 最新を取得
git pull origin develop
# 14. featureブランチ削除
git branch -d feature/add-gitflow-notes
